数据库

主键 一个数据表只包含一个主键，你可以使用主键来查询数据
外键 用于关联两个表
复何健 将多个列作为一个索引键 一般用于复合索引
索引 快速访问数据库特定信息 是对数据库一列或多列的排序 类似书籍目录
参照完整性 数据必须存在

刚安装 密码为空 通过
mysqladmin -u root password “new_password” 修改
登录
mysql -u root -p
show databases;
增加用户
grant all on test(表名？).* ‘yourname’@’localhost’ identified by ‘youpassword’;

进入表
use mysql;
显示表
show tables;
查看表
desc user;

create databases test charset utf8;
查看表创建记录
show create table lin
create table lin(
id int auto_increment,
name char(32) not null,
age int not null,
register_date date not null,
primary ke(id))
insert into lin (name,age,register_data) values(“lin”,22,”2016-3-22”);
select from lin limit 2 offset 1
select from lin where id >3 and age <2
改
update lin set name=”ddd”,age=22 where id=4;
删
delete from lin where name=”ddd”;
排序
select from lin order by id desc; asc 降序升序
分组 计数
select name,count() from lin group by age;
select coalesce(name,”total”),sum(age) as total from lin group by name with rollup;
加列
alter table student add sex enum(“M”,”F”);
删列
alter table lin drop age;
改 是否为空
alter table lin modify sex enum(“M”,”F”) not null;
改字段名
alter table lin change sex gender char(32)not null default “X”
